# if...else if...else...

int main() {
    if (a) {
        return 0;
    } else if (b) {
        return 1;
    } else {
        return 2;
    }
}

==>

Program(
    Function(
        Type, Identifier, CommaSep(),
        Scope(
            Expression(
                Keyword(if), LeftValue(Identifier), Scope(
                    Expression(Keyword(return), RightValue(Number))
                )
            ),
            Expression(
                Keyword(else), Keyword(if), LeftValue(Identifier), Scope(
                    Expression(Keyword(return), RightValue(Number))
                )
            ),
            Expression(
                Keyword(else), Scope(
                    Expression(Keyword(return), RightValue(Number))
                )
            )
        )

# Not equal !=

    if (some_cond_func(a) != 0) {
        return 0;
    }
    return 1;
}

==>

Program(
    Function(
        Type, Identifier, CommaSep(),
        Scope(
            Expression(
                Keyword(if), RightValue(

                    RightValue(Number)
                ),
                Scope(
                    Expression(Keyword(return), RightValue(Number))
                )
            ),
