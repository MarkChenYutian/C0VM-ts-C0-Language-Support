# Single-line @requires

int main()
//@requires true;
{
    return 0;
}

==>

Program(
    Function(
        Type, Identifier, CommaSep(),
        Contract(
            Keyword(requires), Value(Boolean)
        ),
        Scope(
            Line(
                Expression(Keyword(return), Value(Number))
            )
        )
    )
)
